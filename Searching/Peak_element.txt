int peakElement(int arr[], int n)
{
   // Your code here
   int low=0,high=(n-1);
   while(low<=high){
       int mid=(low+high)/2;
       if((mid>0 && mid<(n-1) && arr[mid]>=arr[mid-1] && arr[mid]>=arr[mid+1]) || (mid==0 && arr[mid]>=arr[mid+1]) || mid==(n-1) && arr[mid]>=arr[mid-1]){
           return mid;
       }
       else if(arr[mid+1]>=arr[mid]){
           low=(mid+1);
       }
       else if(arr[mid-1]>=arr[mid]){
           high=(mid-1);
       }
   }
}
